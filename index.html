JS sample

export class Vec2 {
_x Number; // class fields are private by default
_y Number;

constructor(x Number, y Number) {
this._x = x;
this._y = y;
}

get x() Number {
return this._x;
}

get y() Number {
return this._y;
}

length() Number { // class methods are public by default
return Math.sqrt(this._x*this._x + this._y*this._y);
}

normalize() Vec2 {
l := this.length();
return new Vec2(this._x/l, this._y/l);
}
}

// mandelbrot-util.glsl
import {blackbody} from "std/colors";

export vec2 nextGen(vec2 z, vec2 z2, vec2 c) {
float x = z2.x - z2.y + c.x;
float y = 2.0*z.x*z.y + c.y;
return vec2(x, y);
}

export vec4 pickColor(float iter) {
iter = mod(iter, 100.0);
if (iter < 50.0) {
return blackbody(iter/50.0);
} else {
return blackbody((50.0 - iter)/50.0);
}
}
